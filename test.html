func alert(title: String, message: String, viewToAlert: UIViewController, completion: @escaping (UIAlertController) -> Void) {
    let alertController = UIAlertController(title: title, message: message, preferredStyle: UIAlertController.Style.alert)
    let dismissControl = UIControl()
    dismissControl.frame = alertController.view.bounds
    alertController.view.addSubview(dismissControl)
    viewToAlert.present(alertController, animated: true, completion: {
        alertController.view.superview?.isUserInteractionEnabled = true
        completion(alertController)
    })
}
